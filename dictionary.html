function performSearch(query) { const grid = document.getElementById("dictionaryGrid"); const info = document.getElementById("searchResultsInfo"); const countEl = document.getElementById("resultsCount"); const queryEl = document.getElementById("searchQuery"); grid.innerHTML = ""; const q = query.toLowerCase(); const results = Object.values(window.slangDatabase).filter(item => item.term.includes(q) || item.meaning.toLowerCase().includes(q) || item.category.includes(q) ); countEl.textContent = results.length; queryEl.textContent = query; info.style.display = "flex"; if (!results.length) { grid.innerHTML = `

No results found.

`; return; } results.forEach(item => { grid.innerHTML += `
${item.term}
${item.meaning}
${item.category} ⭐ ${item.rating}
`; }); } function setupEnhancedSearch() { document.getElementById("searchButton").addEventListener("click", () => { const val = document.getElementById("dictionarySearch").value.trim(); if (val) performSearch(val); }); document.querySelectorAll(".quick-suggestion").forEach(el => { el.addEventListener("click", () => { const term = el.dataset.term; document.getElementById("dictionarySearch").value = term; performSearch(term); }); }); } function initializeEnhancedDictionary() { performSearch("rizz"); } function showEnhancedTermDetails(term) { const data = window.slangDatabase[term]; if (!data) return; document.getElementById("termModalBody").innerHTML = `

${data.term}

Meaning: ${data.meaning}

Example: ${data.example}

Category: ${data.category}

Safety: ${data.safety}

Rating: ⭐ ${data.rating}

`; document.getElementById("termModal").style.display = "flex"; }
